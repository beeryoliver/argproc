\section{About}
\label{sec:about}

The \pkg{argproc} package provides \pkg{expl3} programming and macro support
for creating user interfaces that take user input into integer, dimen, skip,
and muskip expressions.
The corresponding \eTeX{} primitives---\cs{numexpr}, \cs{dimexpr},
\cs{glueexpr}, and \cs{muexpr}---expand and evaluate the following tokens and
leave the remaining tokens, if any, in the input stream.
This can be problematic if the user provides additional tokens that are not
needed in the expression and can lead to unexpected results and low-level
errors.
This package provides commands that issue an error if the user inputs any
extra trailing tokens that were not processed by the expression.
This package also allows the programmer to append a default unit of
measurement in dimen and skip expressions for user convenience.

\section{Argument processors}
\label{sec:argumentprocessors}

This section documents the argument processors provided by the \pkg{argproc}
package.
For more information on argument processors, see \pkg{usrguide}.%
\footnote{\url{https://ctan.org/pkg/usrguide}}

\begin{function}{\argproc_int:n}
  \begin{syntax}
    \cs{argproc_int:n} \Arg{int expr}
  \end{syntax}
  Argument processor for use in \cs{NewDocumentCommand} and friends.
  Evaluates \meta{int expr} and stores the result in \cs{ProcessedArgument}.
  Issues an error if \meta{int expr} contains any extra trailing tokens that
  were not processed by \meta{int expr}.
\end{function}

\begin{function}{\argproc_dim:nn,\argproc_dim:n}
  \begin{syntax}
    \cs{argproc_dim:nn} \Arg{unit} \Arg{dim expr}
    \cs{argproc_dim:n} \Arg{dim expr}
  \end{syntax}
  Argument processor for use in \cs{NewDocumentCommand} and friends.
  Evaluates \meta{dim expr}, appending a default unit of \meta{unit}, and
  stores the result in \cs{ProcessedArgument}.
  When fully expanded, \meta{unit} must either match a valid unit of
  measurement or expand to nothing.
  Issues an error if \meta{dim expr} contains any extra trailing tokens that
  were not processed by \meta{dim expr}, appending a default unit of
  \meta{unit}.
  \cs{argproc_dim:n} does not append a default unit.
\end{function}

\begin{function}{\argproc_skip:nn,\argproc_skip:n}
  \begin{syntax}
    \cs{argproc_skip:nn} \Arg{unit} \Arg{skip expr}
    \cs{argproc_skip:n} \Arg{skip expr}
  \end{syntax}
  Argument processor for use in \cs{NewDocumentCommand} and friends.
  Evaluates \meta{skip expr}, appending a default unit of \meta{unit}, and
  stores the result in \cs{ProcessedArgument}.
  When fully expanded, \meta{unit} must either match a valid unit of
  measurement or expand to nothing.
  Issues an error if \meta{skip expr} contains any extra trailing tokens that
  were not processed by \meta{skip expr}, appending a default unit of
  \meta{unit}.
  \cs{argproc_skip:n} does not append a default unit.
\end{function}

\begin{function}{\argproc_muskip:n}

\section{Setting integer, dimen, skip, and muskip variables from user input}
\label{sec:programming}


\subsection{Setting integer variables from user input}
\label{subsec:integers}

\begin{function}
  {
    \argproc_int_set_from_user:Nn,
      \argproc_int_set_from_user:cn,
      \argproc_int_set_from_user:NV,
      \argproc_int_set_from_user:cV,
    \argproc_int_gset_from_user:Nn,
      \argproc_int_gset_from_user:cn,
      \argproc_int_gset_from_user:NV,
      \argproc_int_gset_from_user:cV
  }
  \begin{syntax}
    \cs{argproc_int_set_from_user:Nn} \meta{integer} \Arg{int expr}
  \end{syntax}
  Sets \meta{integer} to the value of \meta{int expr}.
  \meta{int expr} should contain the tokens input by the user.
  Issues an error if \meta{int expr} contains any extra trailing tokens that
  were not processed by \meta{int expr}.
\end{function}

\begin{function}
    {
      \argproc_int_add_from_user:Nn,
        \argproc_int_add_from_user:cn,
        \argproc_int_add_from_user:NV,
        \argproc_int_add_from_user:cV,
      \argproc_int_gadd_from_user:Nn,
        \argproc_int_gadd_from_user:cn,
        \argproc_int_gadd_from_user:NV,
        \argproc_int_gadd_from_user:cV
    }
  \begin{syntax}
    \cs{argproc_int_add_from_user:Nn} \meta{integer} \Arg{int expr}
  \end{syntax}
  Identical to \cs{argproc_int_set_from_user:Nn}, but sets \meta{integer}
  equal to its current value plus the result of \meta{int expr}.
\end{function}

\begin{function}
  {
    \argproc_int_sub_from_user:Nn,
      \argproc_int_sub_from_user:cn,
      \argproc_int_sub_from_user:NV,
      \argproc_int_sub_from_user:cV,
    \argproc_int_gsub_from_user:Nn,
      \argproc_int_gsub_from_user:cn,
      \argproc_int_gsub_from_user:NV,
      \argproc_int_gsub_from_user:cV
  }
  \begin{syntax}
    \cs{argproc_int_sub_from_user:Nn} \meta{integer} \Arg{int expr}
  \end{syntax}
  Identical to \cs{argproc_int_set_from_user:Nn}, but sets \meta{integer}
  equal to its current value minus the result of \meta{int expr}.
\end{function}
